// ===================== 只需要改这里 =====================
const (
	lockContractAddr = "0x78B691Fac809bF8Df3Fd6E1B70c11b15Ca9934AB"
	ownerPrivateKey  = "b02fcc641c53f2508696d71fd36fda6128c31cd90621088585683d0553042f42"
	chainID          = 11155111 // Sepolia 链ID
)

// ======================================================

// 防止重复提款
var alreadyWithdraw bool

type SwapEvent struct {
	Sender    common.Address
	EthAmount *big.Int
	MinReturn *big.Int
	To        common.Address
	TxHash    common.Hash
	BlockNum  uint64
	BlockTime time.Time
}

func main() {
	// 1. 连接 Sepolia 测试网节点
	client, err := ethclient.Dial("wss://sepolia.infura.io/ws/v3/4ec7a4f8f43f46e5b51175ff9edbe57f")
	if err != nil {
		log.Fatalf("连接节点失败: %v", err)
	}
	defer client.Close()
	log.Println("✅ Sepolia 节点连接成功")

	// 2. 数据库初始化
	db, err := sql.Open("sqlite3", "monitor.db")
	if err != nil {
		log.Fatalf("打开数据库失败: %v", err)
	}
	defer db.Close()

	_, err = db.Exec(`
	CREATE TABLE IF NOT EXISTS events (
		id INTEGER PRIMARY KEY AUTOINCREMENT,
		tx_hash TEXT UNIQUE,
		sender TEXT,
		eth_amount TEXT,
		min_return TEXT,
		to_addr TEXT,
		block_num INTEGER,
		block_time DATETIME,
		created_at DATETIME DEFAULT CURRENT_TIMESTAMP
	);`)
	if err != nil {
		log.Fatalf("建表失败: %v", err)
	}

	// 你的 Swap 合约地址（Sepolia 上的）
	contractAddr := common.HexToAddress("0x78B691Fac809bF8Df3Fd6E1B70c11b15Ca9934AB")

	eventSig := common.HexToHash("0xbf2ed60bd5b5965d685680c01195c9514e4382e28e3a5a2d2d5244bf59411b93")

	query := ethereum.FilterQuery{
		Addresses: []common.Address{contractAddr},
		Topics:    [][]common.Hash{{eventSig}},
	}

	logs := make(chan types.Log)
	sub, err := client.SubscribeFilterLogs(context.Background(), query, logs)
	if err != nil {
		log.Fatalf("订阅失败: %v", err)
	}
	defer sub.Unsubscribe()

	log.Println("✅ 开始监听 Sepolia 上的 Withdrawal 事件...")
	log.Println("✅ 同时自动监控 Lock 合约解锁时间...")

	// 启动后台协程：定时检查是否可以提款
	go startCheckWithdrawLoop(client)

	for {
		select {
		case err := <-sub.Err():
			log.Fatalf("监听错误: %v", err)
		case vLog := <-logs:
			printWithdrawalEvent(vLog)
		}
	}
}

func printWithdrawalEvent(log types.Log) {
	// 解析数据：2个uint256
	amount := new(big.Int).SetBytes(log.Data[:32])
	when := new(big.Int).SetBytes(log.Data[32:64])

	// 转 ETH
	ethAmount := new(big.Float).Quo(
		new(big.Float).SetInt(amount),
		new(big.Float).SetInt(big.NewInt(1e18)),
	)

	// 时间戳转时间
	blockTime := time.Unix(when.Int64(), 0)

	fmt.Println("\n========================================")
	fmt.Println("🎉 成功监听到 Withdrawal 提款事件！")
	fmt.Println("========================================")
	fmt.Printf("交易哈希: %s\n", log.TxHash.Hex())
	fmt.Printf("合约地址: %s\n", log.Address.Hex())
	fmt.Printf("提款金额: %.6f ETH (%s wei)\n", ethAmount, amount.String())
	fmt.Printf("提款时间: %s\n", blockTime.Format("2006-01-02 15:04:05"))
	fmt.Printf("区块高度: %d\n", log.BlockNumber)
	fmt.Println("========================================\n")
}

// ==============================================
// 定时检查：是否可以调用 withdraw
// ==============================================
func startCheckWithdrawLoop(client *ethclient.Client) {
	ticker := time.NewTicker(3 * time.Second)
	defer ticker.Stop()

	for range ticker.C {
		if alreadyWithdraw {
			continue
		}
		checkAndWithdraw(client)
	}
}

// ==============================================
// 检查 unlockTime + 自动提款
// ==============================================
func checkAndWithdraw(client *ethclient.Client) {
	lockAddr := common.HexToAddress(lockContractAddr)

	// 1. 读取 unlockTime
	unlockTimeBig, err := getUnlockTime(client, lockAddr)
	if err != nil {
		log.Println("读取 unlockTime 失败:", err)
		return
	}

	now := uint64(time.Now().Unix())
	fmt.Printf("[检查提款] 当前时间: %d | 解锁时间: %d\n", now, unlockTimeBig.Uint64())

	if now >= unlockTimeBig.Uint64() {
		fmt.Println("🚀 时间已到，开始自动提款...")

		privKey, err := crypto.HexToECDSA(ownerPrivateKey)
		if err != nil {
			log.Println("私钥错误:", err)
			return
		}

		txHash, err := callWithdraw(client, privKey)
		if err != nil {
			log.Println("提款失败:", err)
			return
		}

		fmt.Printf("✅ 自动提款成功！交易哈希: %s\n", txHash)
		alreadyWithdraw = true
	}
}

// ==============================================
// 读取合约 unlockTime
// ==============================================
func getUnlockTime(client *ethclient.Client, addr common.Address) (*big.Int, error) {
	contract, err := NewLock(addr, client)
	if err != nil {
		return nil, err
	}
	return contract.UnlockTime(nil)
}

// ==============================================
// 调用 withdraw()
// ==============================================
func callWithdraw(client *ethclient.Client, privKey *ecdsa.PrivateKey) (string, error) {
	auth, err := bind.NewKeyedTransactorWithChainID(privKey, big.NewInt(chainID))
	if err != nil {
		return "", err
	}

	// 设置 Gas （Sepolia 必须手动设置）
	auth.GasLimit = 300000
	auth.GasPrice, err = client.SuggestGasPrice(context.Background())
	if err != nil {
		return "", err
	}

	lockContract, err := NewLock(common.HexToAddress(lockContractAddr), client)
	if err != nil {
		return "", err
	}

	tx, err := lockContract.Withdraw(auth)
	if err != nil {
		return "", err
	}

	// 等待上链
	_, err = bind.WaitMined(context.Background(), client, tx)
	if err != nil {
		return "", err
	}

	return tx.Hash().Hex(), nil
}

// ---------------- 工具函数 ----------------
func parseEvent(vLog types.Log, client *ethclient.Client) SwapEvent {
	var evt SwapEvent
	evt.TxHash = vLog.TxHash
	evt.BlockNum = vLog.BlockNumber

	if len(vLog.Topics) >= 4 {
		evt.Sender = common.BytesToAddress(vLog.Topics[1].Bytes())
		evt.EthAmount = new(big.Int).SetBytes(vLog.Topics[2].Bytes())
		evt.To = common.BytesToAddress(vLog.Topics[3].Bytes())
	}

	if len(vLog.Data) >= 32 {
		evt.MinReturn = new(big.Int).SetBytes(vLog.Data[:32])
	}

	block, err := client.BlockByNumber(context.Background(), big.NewInt(int64(vLog.BlockNumber)))
	if err == nil {
		evt.BlockTime = time.Unix(int64(block.Time()), 0)
	} else {
		evt.BlockTime = time.Now()
	}

	return evt
}

func printEvent(evt SwapEvent) {
	ethAmount := new(big.Float).Quo(
		new(big.Float).SetInt(evt.EthAmount),
		new(big.Float).SetInt(big.NewInt(1e18)),
	)

	fmt.Println("\n========================================")
	fmt.Println("🎉 成功监听到 SwapTriggered 事件！")
	fmt.Println("========================================")
	fmt.Printf("交易哈希: %s\n", evt.TxHash.Hex())
	fmt.Printf("发起地址: %s\n", evt.Sender.Hex())
	fmt.Printf("转入ETH: %.6f ETH (%s wei)\n", ethAmount, evt.EthAmount.String())
	fmt.Printf("最小输出: %s\n", evt.MinReturn.String())
	fmt.Printf("接收地址: %s\n", evt.To.Hex())
	fmt.Printf("区块高度: %d\n", evt.BlockNum)
	fmt.Printf("区块时间: %s\n", evt.BlockTime.Format("2006-01-02 15:04:05"))
	fmt.Println("========================================\n")
}

func saveToDB(db *sql.DB, evt SwapEvent) {
	_, err := db.Exec(`
	INSERT OR IGNORE INTO events (
		tx_hash, sender, eth_amount, min_return, to_addr, block_num, block_time
	) VALUES (?, ?, ?, ?, ?, ?, ?)`,
		evt.TxHash.Hex(),
		evt.Sender.Hex(),
		evt.EthAmount.String(),
		evt.MinReturn.String(),
		evt.To.Hex(),
		evt.BlockNum,
		evt.BlockTime.Format("2006-01-02 15:04:05"),
	)
	if err != nil {
		log.Printf("❌ 保存失败: %v", err)
		return
	}
	log.Println("✅ 数据已成功存入数据库")
}
